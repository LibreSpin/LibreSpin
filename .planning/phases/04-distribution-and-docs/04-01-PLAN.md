
       Add the `uninstall()` function immediately before the final dispatcher. It must use this exact list (matching what `install()` writes):
       ```js
       const INSTALL_ITEMS = [
         { path: join('skills', 'librespin-concept'), recursive: true },
         { path: join('agents', 'librespin-concept.md'), recursive: false },
         { path: join('librespin', 'templates'), recursive: true },
       ];

       async function uninstall() {
         console.log(`Uninstalling LibreSpin from ${targetBase}...\n`);
         try {
           for (const item of INSTALL_ITEMS) {
             const target = join(targetBase, item.path);
             await rm(target, { recursive: item.recursive, force: true });
             console.log(`  ✓ Removed ${item.path}`);
           }
           console.log(`\n✓ Uninstall complete.`);
         } catch (err) {
           console.error('\n✗ Uninstall failed:', err.message);
           process.exit(1);
         }
       }
       ```

       Replace the trailing `install();` call with:
       ```js
       if (isUninstall) {
         uninstall();
       } else {
         install();
       }
       ```

    `--uninstall` honours `--local` automatically because both branches reference the same `targetBase` derived earlier in the file (D-08). Use `{ recursive: true, force: true }` so missing paths don't throw — the uninstall is idempotent.

    DO NOT add any npm dependency. DO NOT use third-party packages. `node:fs/promises` only.
